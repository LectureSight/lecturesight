\bundlefile{lecturesight-regiontracker-impl.jar}

The bundle provides the Foreground Region Tracker Service. It tracks regions in the foreground model by mapping labels produced by the connected component analysis in subsequent frames. The service is also capable of filtering regions based on their width, height and total pixel count.

\configproperties

\property{cv.lecturesight.regiontracker.maxnum}{30}{
$[1 \dots \text{MAX\_INT}]$ Maximum number of regions that should be tracked.}

\property{cv.lecturesight.regiontracker.size.min}{50}{
$[1 \dots \text{MAX\_INT}]$ Minimum number of total pixels in a tracked region.}

\property{cv.lecturesight.regiontracker.size.max}{7000}{
$[1 \dots \text{MAX\_INT}]$ Maximum number of total pixels in a tracked region.}

\property{cv.lecturesight.regiontracker.width.min}{15}{
$[1 \dots \text{MAX\_INT}]$ Minimum width of a tracked region.}

\property{cv.lecturesight.regiontracker.width.max}{100}{
$[1 \dots \text{MAX\_INT}]$ Maximum width of a tracked region.}

\property{cv.lecturesight.regiontracker.height.min}{30}{
$[1 \dots \text{MAX\_INT}]$ Minimum height of a tracked region.}

\property{cv.lecturesight.regiontracker.height.max}{200}{
$[1 \dots \text{MAX\_INT}]$ Maximum height of a tracked region.}

\property{cv.lecturesight.regiontracker.debug}{false}{
Whether or not the debug mode for the Region Tracker Service should be active.}
